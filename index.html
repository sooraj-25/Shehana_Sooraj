    const noBtn =

    const funnyText =

      document.getElementById('funnyText');


    const buttons =

      document.getElementById('buttons');


    let repelStrength = 80;


    document.addEventListener('mousemove', (e) => {


      if (!buttons.classList.contains('show'))

        return;


      const rect =

        noBtn.getBoundingClientRect();


      const btnX =

        rect.left + rect.width / 2;


      const btnY =

        rect.top + rect.height / 2;


      const distanceX =

        e.clientX - btnX;


      const distanceY =

        e.clientY - btnY;


      const distance =

        Math.sqrt(

          distanceX * distanceX +

          distanceY * distanceY

        );


      if (distance < 140) {


        const angle =

          Math.atan2(distanceY, distanceX);


        const moveX =

          Math.cos(angle) * repelStrength;


        const moveY =

          Math.sin(angle) * repelStrength;


        noBtn.style.transform =

          `translate(${-moveX}px, ${-moveY}px)`;


        funnyText.innerHTML =

          "Nice try Shehana 😭❤️";


      } else {


        noBtn.style.transform =

          `translate(0px,0px)`;


        funnyText.innerHTML = "";

      }


    });
